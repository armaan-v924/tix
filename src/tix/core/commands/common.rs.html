8/// Build the default branch name for a ticket (with optional description).
9pub fn build_branch_name(config: &Config, ticket_id: &str, description: Option<&String>) -> String {
10    let mut branch_name = format!("{}/{}", config.branch_prefix, ticket_id);
11    if let Some(desc) = description {
12        let sanitized = sanitize_description(desc);
13        if !sanitized.is_empty() {
14            branch_name.push('-');
15            branch_name.push_str(&sanitized);
16        }
17    }
18    branch_name
19}

21/// Sanitize free-form text for inclusion in a git branch name (lowercase, alnum, single hyphens).
22pub fn sanitize_description(input: &str) -> String {
23    let mut result = String::new();
24    let mut last_was_hyphen = true; // Start true to trim leading hyphens
25
26    for c in input.chars() {
27        if c.is_alphanumeric() {
28            result.push(c.to_ascii_lowercase());
29            last_was_hyphen = false;
30        } else if !last_was_hyphen {
31            // Treat everything else (spaces, symbols) as a separator
32            result.push('-');
33            last_was_hyphen = true;
34        }
35    }
36
37    // Trim trailing hyphen if exists
38    if result.ends_with('-') {
39        result.pop();
40    }
41
42    result
43}

45/// Locate the ticket root for a command, either from a provided id or by walking up.
46pub fn locate_ticket_root(ticket: Option<&str>, config: &Config) -> Result<PathBuf> {
47    if let Some(id) = ticket {
48        return Ok(config.tickets_directory.join(id));
49    }
50
51    if let Some(dir) = find_ticket_root_from_cwd() {
52        return Ok(dir);
53    }
54
55    bail!("Could not infer ticket. Run inside a ticket directory or provide --ticket.");
56}
57
58fn find_ticket_root_from_cwd() -> Option<PathBuf> {
59    let mut current = env::current_dir().ok()?;
60    loop {
61        let candidate = current.join(".tix").join("info.toml");
62        if candidate.exists() {
63            return Some(current);
64        }
65
66        if !current.pop() {
67            break;
68        }
69    }
70    None
71}

95    #[test]
96    fn branch_name_includes_description() {
97        let cfg = base_config();
98        let desc = "Short Summary".to_string();
99        let name = build_branch_name(&cfg, "JIRA-1", Some(&desc));
100        assert_eq!(name, "feature/JIRA-1-short-summary");
101    }

103    #[test]
104    fn locate_ticket_root_walks_upwards() {
105        static CWD_LOCK: Mutex<()> = Mutex::new(());
106        let _guard = CWD_LOCK.lock().unwrap();
107
108        let tmp = PathBuf::from(env!("CARGO_MANIFEST_DIR")).join(".tmp-tests-common");
109        let nested = tmp.join("nested/child");
110        fs::create_dir_all(&nested).unwrap();
111        fs::create_dir_all(tmp.join(".tix")).unwrap();
112        fs::write(tmp.join(".tix/info.toml"), "dummy").unwrap();
113
114        let cwd = env::current_dir().unwrap();
115        env::set_current_dir(&nested).unwrap();
116        let found = locate_ticket_root(None, &base_config()).unwrap();
117        env::set_current_dir(cwd).unwrap();
