1//! Destroy a ticket workspace after safety checks.
2
3use crate::core::commands::common::build_branch_name;
4use crate::core::config::Config;
5use crate::core::git;
6use crate::core::ticket::Ticket;
7use anyhow::{Context, Result, anyhow, bail};
8use log::{debug, info, warn};
9use std::env;
10use std::fs;
11use std::path::{Path, PathBuf};
12
13/// Run the destroy command.
14pub fn run(ticket_id: &str, force: bool) -> Result<()> {
15    let config = Config::load()?;
16    let ticket_dir = config.tickets_directory.join(ticket_id);
17
18    if !ticket_dir.exists() {
19        warn!(
20            "Ticket directory {:?} does not exist; nothing to destroy.",
21            ticket_dir
22        );
23        return Ok(());
24    }
25
26    ensure_not_inside(&ticket_dir)?;
27
28    // Try to load metadata to reconstruct branch names; fall back to missing metadata.
29    let ticket_meta = match Ticket::load(&ticket_dir) {
30        Ok(t) => Some(t.metadata),
31        Err(e) => {
32            warn!(
33                "Could not load ticket metadata: {}. Proceeding with limited cleanup.",
34                e
35            );
36            None
37        }
38    };
39
40    let worktree_dirs = worktree_dirs(&ticket_dir);
41    debug!("Found worktree directories: {:?}", worktree_dirs);
42    let aliases_to_prune = aliases_to_prune(&worktree_dirs, ticket_meta.as_ref());
43
44    // Safety checks: ensure clean unless --force
45    if !force {
46        for dir in &worktree_dirs {
47            let is_clean = git::is_clean(dir)
48                .with_context(|| format!("Could not check clean status for {:?}", dir))?;
49            if !is_clean {
50                return Err(anyhow!(
51                    "Worktree at {:?} has uncommitted changes. Use --force to override.",
52                    dir
53                ));
54            }
55        }
56    }
57
58    // Remove directories
59    for dir in &worktree_dirs {
60        if dir.exists() {
61            info!("Removing worktree directory {:?}", dir);
62            fs::remove_dir_all(dir)
63                .with_context(|| format!("Failed to remove worktree directory {:?}", dir))?;
64        }
65    }
66
67    prune_worktrees(&config, ticket_id, ticket_meta.as_ref(), &aliases_to_prune)?;
68
69    info!("Removing ticket directory {:?}", ticket_dir);
70    fs::remove_dir_all(&ticket_dir)
71        .with_context(|| format!("Failed to remove ticket directory {:?}", ticket_dir))?;
72    info!("Destroyed ticket '{}'", ticket_id);
73    Ok(())
74}
75
76fn worktree_dirs(ticket_dir: &Path) -> Vec<PathBuf> {
77    let mut dirs = Vec::new();
78    if let Ok(read) = fs::read_dir(ticket_dir) {
79        for entry in read.flatten() {
80            let path = entry.path();
81            if path.is_dir() && path.file_name().map(|n| n != ".tix").unwrap_or(false) {
82                dirs.push(path);
83            }
84        }
85    }
86    dirs
87}
88
89fn ensure_not_inside(ticket_dir: &Path) -> Result<()> {
90    let current = env::current_dir().context("Failed to get current directory")?;
91    let ticket_canon = ticket_dir
92        .canonicalize()
93        .unwrap_or_else(|_| ticket_dir.to_path_buf());
94    let current_canon = current.canonicalize().unwrap_or(current);
95
96    if current_canon.starts_with(&ticket_canon) {
97        bail!(
98            "Refusing to destroy the ticket while you are inside {:?}",
99            ticket_dir
100        );
101    }
102    Ok(())
103}

105fn aliases_to_prune(
106    worktree_dirs: &[PathBuf],
107    meta: Option<&crate::core::ticket::TicketMetadata>,
108) -> Vec<String> {
109    if let Some(m) = meta {
110        return m.repo_branches.keys().cloned().collect::<Vec<String>>();
111    }
112
113    let mut aliases = Vec::new();
114    for dir in worktree_dirs {
115        if let Some(name) = dir.file_name().and_then(|n| n.to_str()) {
116            aliases.push(name.to_string());
117        }
118    }
119    aliases
120}

122fn prune_worktrees(
123    config: &Config,
124    ticket_id: &str,
125    meta: Option<&crate::core::ticket::TicketMetadata>,
126    aliases: &[String],
127) -> Result<()> {
128    for alias in aliases {
129        let repo_def = match config.repositories.get(alias) {
130            Some(def) => def,
131            None => {
132                warn!(
133                    "Repo alias '{}' not found in config; skipping worktree pruning",
134                    alias
135                );
136                continue;
137            }
138        };
139
140        let branch = meta
141            .and_then(|m| m.repo_branches.get(alias))
142            .cloned()
143            .or_else(|| meta.map(|m| m.branch.clone()))
144            .unwrap_or_else(|| {
145                warn!(
146                    "No stored branch for repo '{}'; deriving branch name for pruning",
147                    alias
148                );
149                build_branch_name(config, ticket_id, meta.and_then(|m| m.description.as_ref()))
150            });
151        let worktree_name = meta
152            .and_then(|m| m.repo_worktrees.get(alias))
153            .cloned()
154            .unwrap_or_else(|| {
155                warn!(
156                    "No stored worktree name for repo '{}'; deriving from branch '{}'",
157                    alias, branch
158                );
159                crate::core::ticket::worktree_name_for_branch(&branch)
160            });
161
162        debug!(
163            "Pruning worktree metadata '{}' in repo {:?}",
164            worktree_name, repo_def.path
165        );
166        git::remove_worktree(&repo_def.path, &worktree_name).with_context(|| {
167            format!(
168                "Failed to prune worktree '{}' for repo '{}' at {:?}",
169                worktree_name, alias, repo_def.path
170            )
171        })?;
172    }
173    Ok(())
174}

184    #[test]
185    fn ensure_not_inside_detects_nested() {
186        static CWD_LOCK: Mutex<()> = Mutex::new(());
187        let _guard = CWD_LOCK.lock().unwrap();
188
189        let tmp = PathBuf::from(env!("CARGO_MANIFEST_DIR")).join(".tmp-tests-destroy");
190        let nested = tmp.join("child");
191        fs::create_dir_all(&nested).unwrap();
192        let cwd = env::current_dir().unwrap();
193        env::set_current_dir(&nested).unwrap();
194        let res = ensure_not_inside(&tmp);
195        env::set_current_dir(cwd).unwrap();
196        assert!(res.is_err());
197    }
