3use anyhow::{Context, Result};

10#[derive(Serialize, Deserialize, Debug, Clone)]
11/// Definition of a registered repository (remote URL and local path).
12pub struct RepoDefinition {
13    /// Remote URL (e.g., `git@github.com:owner/repo.git`).
14    pub url: String,
15    /// Local code path (e.g., `~/code/repo`).
16    pub path: PathBuf,
17}
18
19#[derive(Serialize, Deserialize, Debug, Clone)]
20/// Definition of a registered plugin.
21pub struct PluginDefinition {
22    /// Path to the plugin entrypoint (e.g., `/path/to/plugin.py`).
23    pub entrypoint: PathBuf,
24    /// Optional description shown in listings.
25    #[serde(default)]
26    pub description: String,
27    /// Optional Python interpreter override (e.g., `python3.11`).
28    #[serde(default)]
29    pub python: Option<String>,
30}
31
32#[derive(Serialize, Deserialize, Debug, Default, Clone)]
33/// Global configuration values loaded from `config.toml`.
34pub struct Config {
35    /// Default branch prefix for ticket branches (e.g., `feature`).
36    pub branch_prefix: String,
37    /// Base URL for GitHub SSH/HTTPS clones.
38    pub github_base_url: String,
39    /// Default repository owner used when only a repo name is provided.
40    pub default_repository_owner: String,
41    /// Directory where source repositories live locally.
42    pub code_directory: PathBuf,
43    /// Directory where ticket worktrees are created.
44    pub tickets_directory: PathBuf,
45
46    /// Map of repository aliases to their definitions.
47    pub repositories: HashMap<String, RepoDefinition>,
48
49    /// Map of plugin names to their definitions.
50    #[serde(default)]
51    pub plugins: HashMap<String, PluginDefinition>,
52}
53
54impl Config {
55    /// Load configuration from the OS config directory (e.g., `~/.config/tix/config.toml`).
56    /// Returns `Config::default()` if the file does not exist.
57    pub fn load() -> Result<Self> {
58        let config_path = Self::config_path()?;
59
60        if !config_path.exists() {
61            return Ok(Config::default());
62        }
63
64        let content = std::fs::read_to_string(config_path)?;
65        let config: Config = toml::from_str(&content)?;
66
67        Ok(config)
68    }
69
70    /// Persist the configuration to the OS config directory, creating it if needed.
71    pub fn save(&self) -> Result<()> {
72        let config_path = Self::config_path()?;
73
74        if let Some(parent) = config_path.parent() {
75            std::fs::create_dir_all(parent)?;
76        }
77
78        let toml_string = toml::to_string_pretty(self)?;
79        std::fs::write(config_path, toml_string)?;
80        Ok(())
81    }
82
83    /// Path to the configuration file (e.g., `~/.config/tix/config.toml`).
84    pub fn config_path() -> Result<PathBuf> {
85        if let Some(path) = xdg_config_home_path() {
86            return Ok(path.join("config.toml"));
87        }
88
89        let dirs =
90            ProjectDirs::from("", "", "tix").context("Could not determine config directory")?;
91        Ok(dirs.config_dir().join("config.toml"))
92    }
93}
94
95/// Resolve `$XDG_CONFIG_HOME/tix` when the variable is set and non-empty.
96fn xdg_config_home_path() -> Option<PathBuf> {
97    let dir = env::var_os("XDG_CONFIG_HOME")?;
98    let dir: &Path = dir.as_ref();
99    if dir.as_os_str().is_empty() {
100        return None;
101    }
102    // Validate that the path is absolute and does not contain any parent directory components
103    if !dir.is_absolute()
104        || dir
105            .components()
106            .any(|c| matches!(c, std::path::Component::ParentDir))
107    {
108        return None;
109    }
110    Some(dir.join("tix"))
111}
