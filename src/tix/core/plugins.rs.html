16/// Context passed to plugins (serialized as JSON).
17pub struct PluginContext {
18    /// Registered plugin name.
19    pub plugin_name: String,
20    /// Absolute path to the ticket root.
21    pub ticket_root: PathBuf,
22    /// Working directory when tix was invoked.
23    pub current_working_dir: PathBuf,
24    /// Repo alias for the working directory (if inside a repo worktree).
25    pub current_repo_alias: Option<String>,
26    /// Repo worktree path for the working directory (if inside a repo worktree).
27    pub current_repo_path: Option<PathBuf>,
28    /// Ticket metadata from `.tix/info.toml`.
29    pub ticket: crate::core::ticket::TicketMetadata,
30    /// Full config snapshot at invocation time (read-only by convention).
31    pub config: Config,
32    /// Configured code directory.
33    pub code_directory: PathBuf,
34    /// Configured tickets directory.
35    pub tickets_directory: PathBuf,
36    /// Plugin-specific cache directory.
37    pub plugin_cache_dir: PathBuf,
38    /// Plugin-specific global state directory.
39    pub plugin_state_dir: PathBuf,
40    /// Plugin-specific per-ticket state directory.
41    pub plugin_ticket_state_dir: PathBuf,
42    /// Repository definitions keyed by alias.
43    pub repositories: HashMap<String, RepoDefinition>,
44}

46/// Load plugins from config and return a sorted list.
47pub fn list_plugins() -> Result<Vec<(String, PluginDefinition)>> {
48    let config = Config::load()?;
49    let mut plugins: Vec<_> = config.plugins.into_iter().collect();
50    plugins.sort_by(|a, b| a.0.cmp(&b.0));
51    Ok(plugins)
52}

54/// Entry point for external subcommand routing.
55pub fn run_external(args: Vec<String>) -> Result<()> {
56    if args.is_empty() {
57        bail!("No plugin specified. Run `tix plugins list`.");
58    }
59    let name = &args[0];
60    let plugin_args = &args[1..];
61    run_plugin(name, plugin_args)
62}

64/// Run a registered plugin by name with the provided arguments.
65pub fn run_plugin(name: &str, args: &[String]) -> Result<()> {
66    let config = Config::load()?;
67    let config_path = Config::config_path()?;
68    let working_dir = env::current_dir().context("Failed to resolve current directory")?;
69
70    let plugin = config
71        .plugins
72        .get(name)
73        .cloned()
74        .with_context(|| format!("Unknown plugin '{}'. Run `tix plugins list`.", name))?;
75
76    let entrypoint = resolve_entrypoint(&config_path, &plugin.entrypoint);
77    validate_entrypoint(&entrypoint)?;
78
79    let ticket_root = locate_ticket_root(None, &config)?;
80    let ticket = Ticket::load(&ticket_root)?;
81    let plugin_cache_dir = plugin_cache_dir(name, true)?;
82    let plugin_state_dir = plugin_state_dir(name, true)?;
83    let plugin_ticket_state_dir = plugin_ticket_state_dir(&ticket_root, name, true)?;
84    let (current_repo_alias, current_repo_path) =
85        detect_current_repo(&ticket_root, &ticket.metadata, &working_dir);
86
87    let context = PluginContext {
88        plugin_name: name.to_string(),
89        ticket_root: ticket_root.clone(),
90        current_working_dir: working_dir.clone(),
91        current_repo_alias,
92        current_repo_path,
93        ticket: ticket.metadata,
94        config: config.clone(),
95        code_directory: config.code_directory,
96        tickets_directory: config.tickets_directory,
97        plugin_cache_dir: plugin_cache_dir.clone(),
98        plugin_state_dir: plugin_state_dir.clone(),
99        plugin_ticket_state_dir: plugin_ticket_state_dir.clone(),
100        repositories: config.repositories,
101    };
102
103    let context_file = write_context_file(&ticket_root, &context)?;
104    let context_path = context_file.path().to_path_buf();
105    let project_root = find_uv_project_root(&entrypoint)?;
106
107    let mut command = Command::new("uv");
108    command.arg("run").arg("--project").arg(&project_root);
109    if let Some(python) = plugin.python.as_deref() {
110        command.arg("--python").arg(python);
111    }
112    command
113        .arg("--")
114        .arg("python")
115        .arg("-c")
116        .arg(python_shim())
117        .arg(&entrypoint)
118        .args(args)
119        .current_dir(&ticket_root)
120        .env("TIX_CONTEXT_PATH", &context_path)
121        .env("TIX_TICKET_ROOT", &ticket_root)
122        .env("TIX_PLUGIN_CACHE_DIR", &plugin_cache_dir)
123        .env("TIX_PLUGIN_STATE_DIR", &plugin_state_dir)
124        .env("TIX_PLUGIN_TICKET_STATE_DIR", &plugin_ticket_state_dir);
125
126    let status = command
127        .status()
128        .with_context(|| format!("Failed to run plugin '{}' via uv", name))?;
129
130    if !status.success() {
131        bail!("Plugin '{}' exited with status {}", name, status);
132    }
133
134    Ok(())
135}

137fn resolve_entrypoint(config_path: &Path, entrypoint: &Path) -> PathBuf {
138    if entrypoint.is_absolute() {
139        return entrypoint.to_path_buf();
140    }
141    let base = config_path.parent().unwrap_or_else(|| Path::new("."));
142    base.join(entrypoint)
143}

145fn detect_current_repo(
146    ticket_root: &Path,
147    ticket: &crate::core::ticket::TicketMetadata,
148    working_dir: &Path,
149) -> (Option<String>, Option<PathBuf>) {
150    let mut candidates: Vec<String> = ticket.repo_branches.keys().cloned().collect();
151    for alias in &ticket.repos {
152        if !candidates.contains(alias) {
153            candidates.push(alias.clone());
154        }
155    }
156
157    let mut best_match: Option<(String, PathBuf)> = None;
158    for alias in candidates {
159        let candidate_path = ticket_root.join(&alias);
160        if working_dir.starts_with(&candidate_path) {
161            let replace = match &best_match {
162                Some((_, existing_path)) => {
163                    candidate_path.as_os_str().len() > existing_path.as_os_str().len()
164                }
165                None => true,
166            };
167            if replace {
168                best_match = Some((alias, candidate_path));
169            }
170        }
171    }
172
173    match best_match {
174        Some((alias, path)) => (Some(alias), Some(path)),
175        None => (None, None),
176    }
177}

179fn validate_entrypoint(entrypoint: &Path) -> Result<()> {
180    if !entrypoint.exists() {
181        bail!(
182            "Plugin entrypoint '{}' does not exist",
183            entrypoint.display()
184        );
185    }
186    if entrypoint.is_dir() {
187        bail!(
188            "Plugin entrypoint '{}' is a directory, expected a file",
189            entrypoint.display()
190        );
191    }
192    Ok(())
193}

195fn write_context_file(
196    ticket_root: &Path,
197    context: &PluginContext,
198) -> Result<tempfile::NamedTempFile> {
199    let stamp_dir = ticket_root.join(".tix");
200    std::fs::create_dir_all(&stamp_dir)?;
201    let mut file = tempfile::NamedTempFile::new_in(&stamp_dir)?;
202    serde_json::to_writer(&mut file, context)?;
203    file.flush()?;
204    Ok(file)
205}

207fn python_shim() -> &'static str {
208    r#"
209import json
210import sys
211import importlib.util
212from dataclasses import dataclass
213from typing import Any, Dict, List
214
215@dataclass
216class TixPluginContext:
217    plugin_name: str
218    ticket_root: str
219    current_working_dir: str
220    ticket: Dict[str, Any]
221    config: Dict[str, Any]
222    code_directory: str
223    tickets_directory: str
224    plugin_cache_dir: str
225    plugin_state_dir: str
226    plugin_ticket_state_dir: str
227    repositories: Dict[str, Any]
228
229def load_context(path: str) -> TixPluginContext:
230    with open(path, "r", encoding="utf-8") as f:
231        data = json.load(f)
232    return TixPluginContext(**data)
233
234def load_plugin(entrypoint: str):
235    spec = importlib.util.spec_from_file_location("tix_plugin", entrypoint)
236    if spec is None or spec.loader is None:
237        raise RuntimeError(f"Could not load plugin from {entrypoint}")
238    module = importlib.util.module_from_spec(spec)
239    spec.loader.exec_module(module)
240    return module
241
242def main():
243    if len(sys.argv) < 2:
244        raise RuntimeError("Missing plugin entrypoint")
245    entrypoint = sys.argv[1]
246    argv = sys.argv[2:]
247    ctx_path = os.environ.get("TIX_CONTEXT_PATH")
248    if not ctx_path:
249        raise RuntimeError("TIX_CONTEXT_PATH is not set")
250    ctx = load_context(ctx_path)
251    module = load_plugin(entrypoint)
252    if not hasattr(module, "main"):
253        raise RuntimeError("Plugin must define a main(context, argv) function")
254    module.main(ctx, argv)
255
256if __name__ == "__main__":
257    import os
258    main()
259"#
260}

262fn find_uv_project_root(entrypoint: &Path) -> Result<PathBuf> {
263    let mut current = entrypoint
264        .parent()
265        .context("Plugin entrypoint has no parent directory")?
266        .to_path_buf();
267    loop {
268        let candidate = current.join("pyproject.toml");
269        if candidate.exists() {
270            return Ok(current);
271        }
272        if !current.pop() {
273            break;
274        }
275    }
276    bail!("No pyproject.toml found for plugin; ensure it is a uv project")
277}

279pub fn plugin_cache_root() -> Result<PathBuf> {
280    if let Some(path) = xdg_cache_home_path() {
281        return Ok(path);
282    }
283    let dirs = ProjectDirs::from("", "", "tix").context("Could not determine cache directory")?;
284    Ok(dirs.cache_dir().to_path_buf())
285}

287pub fn plugin_cache_dir(plugin_name: &str, create: bool) -> Result<PathBuf> {
288    let base = plugin_cache_root()?;
289    let sanitized = sanitize_plugin_name(plugin_name);
290    let state_dir = base.join("plugins").join(sanitized);
291    if create {
292        std::fs::create_dir_all(&state_dir)?;
293    }
294    Ok(state_dir)

340fn xdg_cache_home_path() -> Option<PathBuf> {
341    let dir = env::var_os("XDG_CACHE_HOME")?;
342    let dir: &path::Path = dir.as_ref();
343    if dir.as_os_str().is_empty() {
344        return None;
345    }
346    if !dir.is_absolute()
347        || dir
348            .components()
349            .any(|c| matches!(c, path::Component::ParentDir))
350    {
351        return None;
352    }
353    Some(dir.join("tix"))
354}

372fn sanitize_plugin_name(name: &str) -> String {
373    let mut out = String::new();
374    for ch in name.chars() {
375        if ch.is_ascii_alphanumeric() || ch == '-' || ch == '_' {
376            out.push(ch);
377        } else {
378            out.push('_');
379        }
380    }
381    if out.is_empty() {
382        "plugin".to_string()
383    } else {
384        out
385    }
386}
