1//! View or set configuration values.

3use crate::core::config::Config;
4use anyhow::{bail, Context, Result};
5use log::{info, warn};
6use std::path::PathBuf;
7
8/// Set a key to a value or show the current value if `value` is None.
9pub fn run(key: &str, value: Option<&str>) -> Result<()> {
10    let mut config = Config::load()?;
11
12    match key {
13        "branch_prefix" => set_string(&mut config.branch_prefix, key, value)?,
14        "github_base_url" => set_string(&mut config.github_base_url, key, value)?,
15        "default_repository_owner" => set_string(&mut config.default_repository_owner, key, value)?,
16        "code_directory" => set_path(&mut config.code_directory, key, value)?,
17        "tickets_directory" => set_path(&mut config.tickets_directory, key, value)?,
18        other => bail!("Unknown config key '{}'", other),
19    }
20
21    if value.is_some() {
22        config.save().context("Failed to save config")?;
23        info!("Updated '{}'", key);
24    }
25
26    Ok(())
27}
28
29fn set_string(field: &mut String, key: &str, value: Option<&str>) -> Result<()> {
30    if let Some(val) = value {
31        *field = val.to_string();
32    } else {
33        info!("{} = {}", key, field);
34    }
35    Ok(())
36}
37
38fn set_path(field: &mut PathBuf, key: &str, value: Option<&str>) -> Result<()> {
39    if let Some(val) = value {
40        if val.trim().is_empty() {
41            bail!("{} cannot be empty", key);
42        }
43        let expanded = expand_path(val);
44        *field = expanded;
45    } else {
46        info!("{} = {:?}", key, field);
47    }
48    Ok(())
49}
50
51fn expand_path(input: &str) -> PathBuf {
52    if let Some(rest) = input.strip_prefix("~/") {
53        if let Some(home) = home::home_dir() {
54            return home.join(rest);
55        }
56        warn!("~ could not be expanded; using literal path");
57    }
58    PathBuf::from(input)
59}

61#[cfg(test)]
62mod tests {
63    use crate::core::config::Config;
64    use std::collections::HashMap;
65    use std::path::PathBuf;
66
67    fn base_config() -> Config {
68        Config {
69            branch_prefix: "feature".into(),
70            github_base_url: "git@github.com".into(),
71            default_repository_owner: "my-org".into(),
72            code_directory: PathBuf::from("/code"),
73            tickets_directory: PathBuf::from("/tickets"),
74            repositories: HashMap::new(),
75        }
76    }
77
78    #[test]
79    fn set_branch_prefix_updates_value() {
80        let mut config = base_config();
81        run_mut(&mut config, "branch_prefix", Some("hotfix")).unwrap();
82        assert_eq!(config.branch_prefix, "hotfix");
83    }
84
85    #[test]
86    fn path_expansion_supports_home_prefix() {
87        let mut config = base_config();
88        run_mut(&mut config, "code_directory", Some("~/dev")).unwrap();
89        // Either expanded or literal depending on home availability
90        let expected_prefix = home::home_dir()
91            .map(|h| h.join("dev"))
92            .unwrap_or_else(|| PathBuf::from("~/dev"));
93        assert_eq!(config.code_directory, expected_prefix);
94    }
95
96    #[test]
97    fn unknown_key_errors() {
98        let mut config = base_config();
99        assert!(run_mut(&mut config, "unknown", Some("x")).is_err());
100    }
101
102    // helper to invoke run without persisting (bypass load/save)
103    fn run_mut(config: &mut Config, key: &str, value: Option<&str>) -> Result<(), anyhow::Error> {
104        match key {
105            "branch_prefix" => {
106                config.branch_prefix = value.unwrap_or_default().to_string();
107            }
108            "github_base_url" => {
109                config.github_base_url = value.unwrap_or_default().to_string();
110            }
111            "default_repository_owner" => {
112                config.default_repository_owner = value.unwrap_or_default().to_string();
113            }
114            "code_directory" => {
115                config.code_directory = super::expand_path(value.unwrap_or_default());
116            }
117            "tickets_directory" => {
118                config.tickets_directory = super::expand_path(value.unwrap_or_default());
119            }
120            other => return Err(anyhow::anyhow!("Unknown config key '{}'", other)),
121        }
122        Ok(())
123    }
